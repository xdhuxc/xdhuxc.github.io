export function convertArray2Object(source) {
  const len = source.length;
  const target = {};
  for (let i = 0; i < len; i++) {
    target[source[i].key] = source[i].value;
  }
  return target;
}

export function convertObject2Array(source) {
  const target = [];
  for (const k in source) {
    const temp = {
      key: k,
      value: source[k]
    };
    target.push(temp);
  }
  return target;
}
